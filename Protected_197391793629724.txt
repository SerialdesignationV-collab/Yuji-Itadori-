local queue = {}

local isAnimating = false


local function playReplacementAnimation(animationId)

    if isAnimating then

        table.insert(queue, animationId)

        return

    end

   

    isAnimating = true

    local replacementAnimationId = replacementAnimations[tostring(animationId)]

    if replacementAnimationId then

        local AnimAnim = Instance.new("Animation")

        AnimAnim.AnimationId = replacementAnimationId

        local Anim = humanoid:LoadAnimation(AnimAnim)

        Anim:Play()

       

        Anim.Stopped:Connect(function()

            isAnimating = false

            if #queue > 0 then

                local nextAnimationId = table.remove(queue, 1)

                playReplacementAnimation(nextAnimationId)

            end

        end)

    else

        isAnimating = false

    end

end


local function stopSpecificAnimations()

    for _, track in ipairs(humanoid:GetPlayingAnimationTracks()) do

        local animationId = tonumber(track.Animation.AnimationId:match("%d+"))

        if animationIdsToStop[animationId] then

            track:Stop()

        end

    end

end


local function onAnimationPlayed(animationTrack)

    local animationId = tonumber(animationTrack.Animation.AnimationId:match("%d+"))

    if animationIdsToStop[animationId] then

        stopSpecificAnimations()

        animationTrack:Stop()

       

        local replacementAnimationId = replacementAnimations[tostring(animationId)]

        if replacementAnimationId then

            playReplacementAnimation(animationId)

        end

    end

end

local Players = game:GetService("Players")
local player = Players.LocalPlayer

-- Path to the Hotbar GUI
local hotbar = player:WaitForChild("PlayerGui"):WaitForChild("Hotbar") -- Adjust path if needed

-- Table of move names for the 4 hotbar slots
local moveNames = {
    "Move 1: BlackFlash",
    "Move 2: Divergent Barrage",
    "Move 3: Manji Kick",
    "Move 4: Manji Uppercut"
}

-- Function to assign move names to the hotbar slots
local function assignMoveNames()
    for i = 1, 4 do
        local slot = hotbar:FindFirstChild("Slot" .. i) -- Adjust to match your hotbar's slot naming
        if slot and slot:IsA("TextButton") then
            local moveLabel = slot:FindFirstChild("MoveName") -- Replace with your exact label name
            if moveLabel and moveLabel:IsA("TextLabel") then
                moveLabel.Text = moveNames[i] -- Set the move name
                moveLabel.Visible = true -- Ensure it's visible
            else
                -- Create a TextLabel if it doesn't exist
                moveLabel = Instance.new("TextLabel")
                moveLabel.Name = "MoveName"
                moveLabel.Parent = slot
                moveLabel.Text = moveNames[i]
                moveLabel.Size = UDim2.new(1, 0, 0.2, 0) -- Adjust size as needed
                moveLabel.Position = UDim2.new(0, 0, 1, 0) -- Position below the slot
                moveLabel.BackgroundTransparency = 1 -- Make background transparent
                moveLabel.TextColor3 = Color3.new(1, 1, 1) -- White text
                moveLabel.Font = Enum.Font.SourceSans -- Font style
                moveLabel.TextScaled = true -- Scale text to fit
                moveLabel.Visible = true
            end
        end
    end
end

-- Assign move names when the script runs
assignMoveNames()

-- Reassign move names if hotbar updates
hotbar.ChildAdded:Connect(assignMoveNames)
hotbar.ChildRemoved:Connect(assignMoveNames)
